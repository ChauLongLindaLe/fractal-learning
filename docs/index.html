Here is an example of the new component

Here’s the rendered version

<!-- Start: component HTML -->
<div id="js-alert" class="alert alert--{{ modifier }}">
    <h2 class="alert__title">{{ heading }}</h2>

    <p class="alert__body">This component is using Handlebar templating</p>

    <p class="alert__body">
        <img src="{{ path '/img/warning-triangle.svg' }}" width="25" height="25" alt="Warning">

        {{ text }}
    </p>

    <p id="js-alert__message" class="alert__message alert_message--hidden">
        This is hidden text. Lorem ipsum dolor, sit amet consectetur adipisicing elit. Mollitia, ducimus!
    </p>

    <p class="alert__nav">
        <button id="js-alert__button" class="alert__button">Toggle hidden paragraph</button>
    </p>
</div>
<!-- End: component HTML -->

<!-- Start: component scripts -->
<script>
    document.getElementById("js-alert__button").addEventListener("click", function () {
        document.getElementById("js-alert__message").classList.toggle("is-active");
    });
</script>
<!-- End: component scripts -->
